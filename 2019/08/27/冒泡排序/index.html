package com.example.study.sort;

import org.springframework.util.Assert;
import org.springframework.util.StopWatch;

import java.util.stream.Stream;

/**
 *
 * @Description : 主要用作对bubbleSort的探索，这里列出了算法，并且根据改进。
 *
 * 算法参考思想：《小灰的漫画算法》
 *
 * @author : sheen.lee
 * @date : 19:15 2019/8/27
 *
 */
public class Bubble {

    /**
     * 这里是冒泡排序，
     *
     * @param array
     * @param <T>
     * @return
     * @version 1.0
     */
    public static <T extends Number & Comparable> T[] bubbleSort1(T[] array) {
        Assert.notNull(array, "params must not be null");
        Assert.notEmpty(array, "params must not be empty");

        for (int i = 0; i < array.length - 1; i++) {
            for (int j = 0; j < array.length - 1 - i; j++) {
                T temp;
                if (array[j].compareTo(array[j + 1]) > 0) {
                    temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                }
            }
        }
        return array;

    }


    /**
     * 这里是冒泡排序，
     * 这里用一个标识去判断是否已经是完整的排序了。
     *
     * @param array
     * @param <T>
     * @return
     * @version 2.0
     */
    public static <T extends Number & Comparable> T[] bubbleSort2(T[] array) {
        Assert.notNull(array, "params must not be null");
        Assert.notEmpty(array, "params must not be empty");

        for (int i = 0; i < array.length - 1; i++) {
            boolean swap = false;
            for (int j = 0; j < array.length - 1 - i; j++) {
                T temp;
                if (array[j].compareTo(array[j + 1]) > 0) {
                    temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    swap = true;
                }
            }

            if (!swap) {
                break;
            }
        }
        return array;

    }


    /**
     * 这里是冒泡排序，
     * 这里用一个标识去判断是否已经是完整的排序了。
     * 这里去判断是否后边的数据已经排好序，下次不再比较后边的数据。
     *
     * @param array
     * @param <T>
     * @return
     * @version 3.0
     */
    public static <T extends Number & Comparable> T[] bubbleSort3(T[] array) {
        Assert.notNull(array, "params must not be null");
        Assert.notEmpty(array, "params must not be empty");

        int sortedBorder = array.length - 1;
        for (int i = 0; i < array.length - 1; i++) {
            boolean swap = false;
            int sortedBorder1 = sortedBorder;
            for (int j = 0; j < sortedBorder1; j++) {
                T temp;
                if (array[j].compareTo(array[j + 1]) > 0) {
                    temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    swap = true;
                    sortedBorder = j;
                }
            }

            if (!swap) {
                break;
            }
        }
        return array;

    }
