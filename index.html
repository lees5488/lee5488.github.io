package com.example.study.sort;

import org.springframework.util.Assert;
import org.springframework.util.StopWatch;

/**
 *
 * @Description : 主要用作对bubbleSort的探索，这里列出了算法，并且根据改进。
 *
 * 算法参考思想：《小灰的漫画算法》
 *
 * @author : sheen.lee
 * @date : 19:15 2019/8/27
 *

public class Bubble {

    /**
     * 这里是冒泡排序，
     *
     * @param array
     * @param <T>
     * @return
     * @version 1.0
     */
    public static <T extends Number & Comparable> T[] bubbleSort1(T[] array) {
        Assert.notNull(array, "params must not be null");
        Assert.notEmpty(array, "params must not be empty");

        for (int i = 0; i < array.length - 1; i++) {
            for (int j = 0; j < array.length - 1 - i; j++) {
                T temp;
                if (array[j].compareTo(array[j + 1]) > 0) {
                    temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                }
            }
        }
        return array;

    }


    /**
     * 这里是冒泡排序，
     * 这里用一个标识去判断是否已经是完整的排序了。
     *
     * @param array
     * @param <T>
     * @return
     * @version 2.0
     */
    public static <T extends Number & Comparable> T[] bubbleSort2(T[] array) {
        Assert.notNull(array, "params must not be null");
        Assert.notEmpty(array, "params must not be empty");

        for (int i = 0; i < array.length - 1; i++) {
            boolean swap = false;
            for (int j = 0; j < array.length - 1 - i; j++) {
                T temp;
                if (array[j].compareTo(array[j + 1]) > 0) {
                    temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    swap = true;
                }
            }

            if (!swap) {
                break;
            }
        }
        return array;

    }


    /**
     * 这里是冒泡排序，
     * 这里用一个标识去判断是否已经是完整的排序了。
     * 这里去判断是否后边的数据已经排好序，下次不再比较后边的数据。
     *
     * @param array
     * @param <T>
     * @return
     * @version 3.0
     */
    public static <T extends Number & Comparable> T[] bubbleSort3(T[] array) {
        Assert.notNull(array, "params must not be null");
        Assert.notEmpty(array, "params must not be empty");

        int sortedBorder = array.length - 1;
        for (int i = 0; i < array.length - 1; i++) {
            boolean swap = false;
            int sortedBorder1 = sortedBorder;
            for (int j = 0; j < sortedBorder1; j++) {
                T temp;
                if (array[j].compareTo(array[j + 1]) > 0) {
                    temp = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = temp;
                    swap = true;
                    sortedBorder = j;
                }
            }

            if (!swap) {
                break;
            }
        }
        return array;

    }

最近遇到問題需要使用正則表達式，以下是總結的部分知識

转载自：uTank-木头
首先介绍一下“回车”（carriage return,’\r’）和“换行”（line feed,’\n’）这两个概念的来历和区别。
在计算机还没有出现之前，有一种叫做电传打字机（Teletype Model 33）的玩意，每秒钟可以打10个字符。
但是它有一个问题，就是打完一行换行的时候，要用去0.2秒，正好可以打两个字符。
要是在这0.2秒里面，又有新的字符传过来，那么这个字符将丢失。于是，研制人员想了个办法解决这个问题，就是在每行后面加两个表示结束的字符。
一个叫做“回车”，告诉打字机把打印头定位在左边界；另一个叫做“换行”，告诉打字机把纸向下移一行。
这就是“换行”和“回车”的来历，从它们的英语名字上也可以看出一二。
